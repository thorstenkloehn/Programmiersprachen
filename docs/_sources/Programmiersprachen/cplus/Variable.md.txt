## Ganzezahlen

```
#include <iostream>
#include <cstdint>

int main() {

    int8_t tho = 100;
    int16_t tho16;
    int32_t tho32;
    int64_t tho64;
    uint8_t utho = 111;
    uint16_t utho16;
    uint32_t hhhh;
    uint64_t ghjg;

std::cout << int(tho) << std::endl;
std::cout << int(utho);
}
return 0
}
```
